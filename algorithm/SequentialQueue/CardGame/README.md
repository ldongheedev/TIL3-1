/* 
   카드게임 프로젝트

   문제: N장의 카드가 있을 때, 다음 동작을 반복하여 마지막에 남는 카드를 찾기
   1. 가장 위의 카드를 버린다
   2. 새로운 가장 위의 카드를 가장 아래로 옮긴다

   예제:
   N = 4일 때, 마지막에 남는 카드는 4
*/

#include <stdio.h>
#include <stdlib.h>

int main() {
    // N = 4 예제 실행
    printf("=== 카드게임 시뮬레이션 ===\n\n");
    printf("N = 4일 때:\n");
    printf("초기: 1, 2, 3, 4\n");
    printf("1을 버림 -> 2, 3, 4\n");
    printf("2를 아래로 -> 3, 4, 2\n");
    printf("3을 버림 -> 4, 2\n");
    printf("4를 아래로 -> 2, 4\n");
    printf("2를 버림 -> 4\n");
    printf("\n결과: 마지막 카드는 4\n");

    return 0;
}
